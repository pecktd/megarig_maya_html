    refnodes = sorted(mc.ls(type="reference"), reverse=True)

    if not refnodes:
        return True

    for rfn in refnodes:
        if rfn == "sharedReferenceNode":
            continue
        try:
            fn = mc.referenceQuery(rfn, filename=True)
            mc.file(fn, importReference=True)
        except RuntimeError:
            mc.lockNode(rfn, l=0)
            mc.delete(rfn)

[docs]def flatten_all_nodes_in_namesapce(ns: str) -> bool:
    nodes = sorted(mc.ls("{}:*".format(ns), l=True), key=len, reverse=True)
    mc.namespace(set=":")

    if not nodes:
        return True

    for node in nodes:
        if not mc.objExists(node):
            continue
        lock_state = mc.lockNode(node, q=True)[0]
        if lock_state:
            mc.lockNode(node, l=False)
        newname = node
        try:
            newname = mc.rename(node, node.split(":")[-1])
        except:
            pass
        mc.lockNode(newname, l=lock_state)

    return True




[docs]def remove_unused_influences(skinnode: str):
    infs = mc.skinCluster(skinnode, q=True, inf=True)
    wtinfs = mc.skinCluster(skinnode, q=True, wi=True)

    if not infs:

            continue
        if mc.getAttr("{}.io".format(mesh)) and not mc.listConnections(mesh):

[docs]def get_all_mesh_xforms():
    meshes = mc.ls(type="mesh", l=True)
    meshxforms = []

    if not meshes:
        return meshxforms

    for mesh in meshes:
        pars = mc.listRelatives(mesh, p=True, f=True)
        if mc.referenceQuery(pars[0], isNodeReferenced=True):
            continue
        if pars[0] in meshxforms:
            continue
        meshxforms.append(pars[0])

    return meshxforms

    if not shapes:
        return None

    if len(shapes) == 1:
        return shapes[0]

    for shape in shapes:
        if mc.getAttr("{}.intermediateObject".format(shape)):
            connections = mc.listConnections(
                "{}.inMesh".format(shape), s=True, d=False

            if not connections:
                return shape

[docs]def clean_all_meshes() -> bool:
    """Removes all color sets, unlock all normals and smooth all edges."""
    for mesh in get_all_editable_shapes():
        intermediate = mc.getAttr("{}.intermediateObject".format(mesh))

        if intermediate:
            mc.setAttr("{}.intermediateObject".format(mesh), False)

        color_sets = mc.polyColorSet(mesh, q=True, allColorSets=True)
        if color_sets:
            for colorset in color_sets:
                mc.polyColorSet(mesh, delete=True, colorSet=colorset)

        mc.polyNormalPerVertex(mesh, ufn=True)
        mc.polySoftEdge(mesh, a=180, ch=False)

        if intermediate:
            mc.setAttr("{}.intermediateObject".format(mesh), True)

    return True




[docs]def remove_all_unknowns() -> bool:
    """Recursively removes unknown nodes from current scene."""
    unkns = mc.ls(type="unknown")

    if not unkns:
        return True

    for unkn in unkns:
        if mc.objExists(unkn):
            try:
                mc.delete(unkn)
                print("{} has been removed.".format(unkn))
            except RuntimeError:
                pass
    remove_all_unknowns()

    return True




[docs]def remove_all_shaders():
    exceptions = ("initialShadingGroup", "initialParticleSE")

    for se in mc.ls(type="shadingEngine"):
        if se in exceptions:
            continue

        shaders = mc.listConnections("{}.surfaceShader".format(se))
        if shaders:
            try:
                mc.delete(shaders[0])
                print("{} has been deleted.".format(shaders[0]))
            except RuntimeError:
                print("{} cannot be deleted.".format(shaders[0]))
        try:
            mc.delete(se)
            print("{} has been deleted.".format(se))
        except RuntimeError:
            print("{} cannot be deleted.".format(se))

    for mesh in mc.ls(type="mesh"):
        mc.select(mesh, r=True)
        mc.sets(e=True, forceElement="initialShadingGroup")

[docs]def cleanup_scene():
    for node in DELETE_NODES:
        if not mc.objExists(node):
            continue
        for chd in mc.listRelatives(node, ad=True, type="transform"):
            if not mc.objExists(chd):
                continue
            if mc.referenceQuery(chd, inr=True):
                rn = mc.referenceQuery(chd, rfn=True, tr=True)
                fn = mc.referenceQuery(rn, filename=True)
                mc.file(fn, rr=True)

    import_all_references()

    node_types = ("script", "renderLayer", "displayLayer")
    exceptions = ("defaultRenderLayer", "defaultLayer")

    for nodetype in node_types:
        nodes = mc.ls(type=nodetype)

        if not nodes:
            continue

        for node in nodes:
            if node in exceptions:
                continue
            try:
                mc.delete(node)
            except RuntimeError:
                print("{} cannot be deleted.".format(node))

    for node in DELETE_NODES:
        if mc.objExists(node):
            mc.delete(node)

    flatten_all_namespaces()
    remove_all_ng_skintools_nodes()
    remove_all_unused_influences()
    remove_all_unused_intermediate_shapes()
    remove_all_anim_curves()
    clean_all_meshes()
    remove_all_cameras()
