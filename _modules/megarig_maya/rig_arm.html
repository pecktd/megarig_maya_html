[docs]class ArmRig(rig_base.BaseRig):
    def __init__(
        self,
        upper_arm_guide: str,
        elbow_guide: str,
        wrist_guide: str,
        hand_guide: str,
        mod_desc: str,
        side: mrc.Side,
        parent: mrc.Dag,
        skin_parent: mrc.Dag,
        still_parent: mrc.Dag,
        fk_spaces_attrnames: Union[Tuple[Tuple[mrc.Dag, str]], None],
        ik_spaces_attrnames: Union[Tuple[Tuple[mrc.Dag, str]], None],
        vol_pres: bool,
    ):
        super(ArmRig, self).__init__("arm", mod_desc, side)
        if parent:
            self.meta.set_parent_at(parent)

        self.upper_arm_guide = mrc.Dag(upper_arm_guide)
        self.elbow_guide = mrc.Dag(elbow_guide)
        self.wrist_guide = mrc.Dag(wrist_guide)
        self.hand_guide = mrc.Dag(hand_guide)

        self.arm_len = om.MVector(
            (self.elbow_guide.world_vec - self.upper_arm_guide.world_vec)
            + (self.wrist_guide.world_vec - self.elbow_guide.world_vec)
        ).length()

        # Main Groups
        self.scale_grp = self.init_dag(mrc.Null(), self.side, ("sc",), "grp")
        self.scale_grp.snap(self.upper_arm_guide)
        self.scale_grp.set_parent(self.meta)

        self.still_grp = self.init_dag(
            mrc.Null(), self.side, ("still",), "grp"
        )
        if still_parent:
            self.still_grp.set_parent(still_parent)

        # Main Joints
        self.upper_arm_jnt = self.init_dag(
            mru.create_joint_at(self.upper_arm_guide),
            self.side,
            ("up",),
            "jnt",
        )
        self.elbow_jnt = self.init_dag(
            mru.create_joint_at(self.elbow_guide),
            self.side,
            ("elbow",),
            "jnt",
        )
        self.wrist_jnt = self.init_dag(
            mru.create_joint_at(self.wrist_guide),
            self.side,
            ("wrist",),
            "jnt",
        )
        self.hand_jnt = self.init_dag(
            mru.create_joint_at(self.hand_guide),
            self.side,
            ("hand",),
            "jnt",
        )
        self.hand_jnt.set_parent(self.wrist_jnt)
        self.hand_jnt.ssc = False
        self.wrist_jnt.set_parent(self.elbow_jnt)
        self.elbow_jnt.set_parent(self.upper_arm_jnt)
        self.elbow_jnt.ssc = False
        if skin_parent:
            self.upper_arm_jnt.set_parent(skin_parent)

        # Proxy Joints
        self.upper_arm_pxj = self.init_dag(
            mru.create_joint_at(self.upper_arm_jnt),
            self.side,
            ("up",),
            "pxJnt",
        )
        self.elbow_pxj = self.init_dag(
            mru.create_joint_at(self.elbow_jnt),
            self.side,
            ("elbow",),
            "pxJnt",
        )
        self.wrist_pxj = self.init_dag(
            mru.create_joint_at(self.wrist_jnt),
            self.side,
            ("wrist",),
            "pxJnt",
        )
        self.hand_pxj = self.init_dag(
            mru.create_joint_at(self.hand_jnt),
            self.side,
            ("hand",),
            "pxJnt",
        )
        self.upper_arm_pxj_zr = self.init_dag(
            mru.add_zero_group(self.upper_arm_pxj),
            self.side,
            ("up", "pxJnt"),
            "zr",
        )

        self.hand_pxj.set_parent(self.wrist_pxj)
        self.hand_pxj.ssc = False
        self.wrist_pxj.set_parent(self.elbow_pxj)
        self.elbow_pxj.set_parent(self.upper_arm_pxj)
        self.upper_arm_pxj_zr.set_parent(self.scale_grp)

        self.upper_arm_nrj, _, _, _, _, _ = self.add_non_roll_joint(
            self.upper_arm_pxj, "y", ("up",)
        )
        self.wrist_nrj, _, _, _, _, _ = self.add_non_roll_joint(
            self.wrist_pxj, "y", ("wrist",)
        )

        self.upper_arm_roll_attr = self.upper_arm_nrj.attr("twist")
        self.wrist_roll_attr = self.wrist_nrj.attr("twist")

        if self.side == mrc.Side.right:

            def _add_inverse_roll(attr: mrc.Attr, name: str) -> mrc.Attr:
                mult = self.init_node(
                    mrc.create("multDoubleLinear"),
                    self.side,
                    (name, "tw"),
                    "inv",
                )
                attr >> mult.attr("i1")
                mult.attr("i2").v = -1

                return mult.attr("o")

            self.upper_arm_roll_attr = _add_inverse_roll(
                self.upper_arm_nrj.attr("twist"), "up"
            )
            self.wrist_roll_attr = _add_inverse_roll(
                self.wrist_nrj.attr("twist"), "wrist"
            )

        # Main Controller
        (
            self.main_ctrl,
            self.main_ctrl_ofst,
            self.main_ctrl_ori,
            self.main_ctrl_zr,
        ) = self.add_controller(
            self.side,
            ("",),
            mrc.Cp.stick,
            self.color,
            ("t", "r", "s", "v", "ro"),
        )
        self.main_ctrl.rotate_shape((0, 0, 90))
        self.main_ctrl_zr.snap(self.scale_grp)
        self.main_ctrl_zr.set_parent(self.meta)
        self.main_ctrl_shape = self.main_ctrl.get_shape()

        self.arm_sc_attr = self.main_ctrl.add(ln="armScale", dv=1, k=True)
        self.hand_sc_attr = self.main_ctrl.add(ln="handScale", dv=1, k=True)

        self.arm_sc_attr >> self.scale_grp.attr("sx")
        self.arm_sc_attr >> self.scale_grp.attr("sy")
        self.arm_sc_attr >> self.scale_grp.attr("sz")
        self.arm_sc_attr >> self.upper_arm_jnt.attr("sx")
        self.arm_sc_attr >> self.upper_arm_jnt.attr("sy")
        self.arm_sc_attr >> self.upper_arm_jnt.attr("sz")

        fkik_attr = self.main_ctrl.add(ln="fkIk", min=0, max=1, k=True)
        self.fkik_rev = self.init_node(
            mrc.create("reverse"), self.side, ("fk", "ik"), "rev"
        )

        # FK
        self.fk = rig_fk.LimbFkRig(
            guides=(self.upper_arm_guide, self.elbow_guide, self.wrist_guide),
            side=self.side,
            mod_name="arm_fk",
            mod_desc="",
            joint_type="fkJnt",
            ctrl_shape=mrc.Cp.double_circle,
            color=self.color,
            parent=self.scale_grp,
            spaces_attrnames=fk_spaces_attrnames,
        )
        self.fk.ctrls[1].lhattrs("ry", "rz", "ro")

        self.hand_fk_jnt = self.init_dag(
            mru.create_joint_at(self.hand_guide),
            self.side,
            ("fk", "4"),
            "fkJnt",
        )
        self.hand_fk_jnt.ssc = False
        self.hand_fk_jnt.set_parent(self.fk.jnts[-1])

        # FK
        self.ik = rig_stretchy_ik.StretchyIk(
            guides=(
                self.upper_arm_guide,
                self.elbow_guide,
                self.wrist_guide,
                self.hand_guide,
            ),
            ee_index=2,
            pole_vector_offset=0.5,
            actual_length=True,
            spaces_attrnames=ik_spaces_attrnames,
            mod_name="arm_ik",
            desc="",
            ctrl_shape=mrc.Cp.cube,
            ctrl_color=self.color,
            joint_type="ikJnt",
            side=self.side,
            parent=self.scale_grp,
        )

        self.hand_ikh = self.init_dag(
            mrc.ikrp(sj=self.ik.jnts[2], ee=self.ik.jnts[3]),
            self.side,
            ("hand",),
            "ikh",
        )

        self.hand_piv = self.init_dag(
            mrc.Null(), self.side, ("ik", "hand"), "piv"
        )
        self.hand_piv.set_parent_at(self.ik.ctrl)
        self.hand_ikh.set_parent(self.ik.ctrl)
        self.hand_ikh.hide = True

        # FK/IK Blend
        fkik_attr >> self.fkik_rev.attr("ix")
        fkik_attr >> self.ik.meta.attr("v")
        self.fkik_rev.attr("ox") >> self.fk.meta.attr("v")

        self.init_node(
            mru.tr_blend(
                self.fk.jnts[0],
                self.ik.jnts[0],
                self.upper_arm_pxj,
                fkik_attr,
            ),
            self.side,
            ("fk", "ik", "up", "arm"),
            "blend",
        )
        self.init_node(
            mru.tr_blend(
                self.fk.jnts[1],
                self.ik.jnts[1],
                self.elbow_pxj,
                fkik_attr,
            ),
            self.side,
            ("fk", "ik", "knee"),
            "blend",
        )
        self.init_node(
            mru.tr_blend(
                self.fk.jnts[2],
                self.ik.jnts[2],
                self.wrist_pxj,
                fkik_attr,
            ),
            self.side,
            ("fk", "ik", "wrist"),
            "blend",
        )
        self.init_node(
            mru.tr_blend(
                self.hand_fk_jnt,
                self.ik.jnts[3],
                self.hand_pxj,
                fkik_attr,
            ),
            self.side,
            ("fk", "ik", "hand"),
            "blend",
        )

        mrc.parcons(self.upper_arm_nrj, self.upper_arm_jnt)
        mrc.parcons(self.elbow_pxj, self.elbow_jnt)
        mrc.parcons(self.wrist_pxj, self.wrist_jnt)
        mrc.parcons(self.hand_pxj, self.hand_jnt)

        # Hand Node
        self.hand_spc = self.init_dag(
            mrc.Null(), self.side, ("hand", "spc"), "grp"
        )
        self.hand_spc.set_parent(self.scale_grp)
        mrc.parcons(self.hand_pxj, self.hand_spc)

        # Hand scale
        self.hand_sc_attr >> self.fk.jnts[2].attr("sx")
        self.hand_sc_attr >> self.fk.jnts[2].attr("sy")
        self.hand_sc_attr >> self.fk.jnts[2].attr("sz")
        self.hand_sc_attr >> self.fk.ctrl_ories[2].attr("sx")
        self.hand_sc_attr >> self.fk.ctrl_ories[2].attr("sy")
        self.hand_sc_attr >> self.fk.ctrl_ories[2].attr("sz")

        self.hand_sc_attr >> self.ik.jnts[2].attr("sx")
        self.hand_sc_attr >> self.ik.jnts[2].attr("sy")
        self.hand_sc_attr >> self.ik.jnts[2].attr("sz")
        self.hand_sc_attr >> self.hand_piv.attr("sx")
        self.hand_sc_attr >> self.hand_piv.attr("sy")
        self.hand_sc_attr >> self.hand_piv.attr("sz")

        self.hand_sc_attr >> self.wrist_pxj.attr("sx")
        self.hand_sc_attr >> self.wrist_pxj.attr("sy")
        self.hand_sc_attr >> self.wrist_pxj.attr("sz")

        self.hand_sc_attr >> self.hand_spc.attr("sx")
        self.hand_sc_attr >> self.hand_spc.attr("sy")
        self.hand_sc_attr >> self.hand_spc.attr("sz")

        self.wrist_pxj.attr("s") >> self.wrist_jnt.attr("s")

        # Ribbon IK
        (
            self.mid_ctrl,
            self.mid_ctrl_ofst,
            self.mid_ctrl_ori,
            self.mid_ctrl_zr,
        ) = self.add_controller(
            self.side, ("mid",), mrc.Cp.plus, self.color, ("r", "s", "v", "ro")
        )
        self.mid_ctrl_zr.set_parent(self.scale_grp)
        mrc.parcons(self.elbow_jnt, self.mid_ctrl_zr)

        self.upper_arm_ribik = rig_ribbon_ik.RibbonIk(
            root=self.upper_arm_jnt,
            end=self.mid_ctrl,
            root_tg=self.upper_arm_nrj,
            end_tg=self.mid_ctrl,
            parent_tg=self.upper_arm_nrj,
            jnt_amt=5,
            side=self.side,
            mod_name="ik_up_arm",
            desc="",
            ctrl_shp=mrc.Cp.double_circle,
            dtl_ctrl_shp=mrc.Cp.none,
            ctrl_clr=self.dark_color,
            dtl_clr=self.color,
            parent=self.scale_grp,
            skin_parent=self.upper_arm_jnt,
            still_parent=self.still_grp,
        )
        self.upper_arm_roll_attr >> self.upper_arm_ribik.end_twist

        self.forearm_ribik = rig_ribbon_ik.RibbonIk(
            root=self.elbow_jnt,
            end=self.wrist_jnt,
            root_tg=self.mid_ctrl,
            end_tg=self.wrist_nrj,
            parent_tg=self.elbow_pxj,
            jnt_amt=5,
            side=self.side,
            mod_name="ik_forearm",
            desc="",
            ctrl_shp=mrc.Cp.double_circle,
            dtl_ctrl_shp=mrc.Cp.none,
            ctrl_clr=self.dark_color,
            dtl_clr=self.color,
            parent=self.scale_grp,
            skin_parent=self.elbow_jnt,
            still_parent=self.still_grp,
        )
        self.wrist_roll_attr >> self.forearm_ribik.end_twist

        # Cleanup
        self.upper_arm_pxj.hide = True
        self.upper_arm_nrj.hide = True
        self.fk.jnts[0].hide = True
        self.ik.jnts[0].hide = True

        self.main_ctrl.attr("fkIk").v = 0
        self.fk.ctrls[0].attr("orientTo").v = 2
        self.ik.ctrl.attr("parentTo").v = 1

        self.add_nodes_to_set(
            self.upper_arm_ribik.ribbon.jnts
            + self.forearm_ribik.ribbon.jnts
            + [self.wrist_jnt, self.hand_jnt],
            mrc.SKIN_SET,
        )

        # Add volume preservation
        volpres_ofst = 0.05 * self.arm_len
        if vol_pres:
            visattr = self.main_ctrl.get_shape().add(
                ln="volPresVis", min=0, max=1, dv=0, k=True
            )
            self.upper_arm_volpres = rig_volpres.VolPresRig(
                target=self.upper_arm_jnt,
                target_pos=self.upper_arm_jnt,
                parent=self.meta,
                offset=volpres_ofst,
                volume_joints=(True, True, False, False, True, True),
                mod_name="upper_arm_volpres",
                desc="",
                side=self.side,
                joint_type="jnt",
            )
            self.elbow_volpres = rig_volpres.VolPresRig(
                target=self.elbow_jnt,
                target_pos=self.mid_ctrl,
                parent=self.meta,
                offset=volpres_ofst,
                volume_joints=(True, True, False, False, True, True),
                mod_name="elbow_volpres",
                desc="",
                side=self.side,
                joint_type="jnt",
            )
            self.wrist_volpres = rig_volpres.VolPresRig(
                target=self.wrist_jnt,
                target_pos=self.wrist_jnt,
                parent=self.meta,
                offset=volpres_ofst,
                volume_joints=(True, True, False, False, True, True),
                mod_name="wrist_volpres",
                desc="",
                side=self.side,
                joint_type="jnt",
            )

            visattr >> self.upper_arm_volpres.meta.attr("v")
            visattr >> self.elbow_volpres.meta.attr("v")
            visattr >> self.wrist_volpres.meta.attr("v")

            self.add_nodes_to_set(
                self.upper_arm_volpres.jnts
                + self.elbow_volpres.jnts
                + self.wrist_volpres.jnts,
                mrc.SKIN_SET,
            )

        # RO
        ctrl: mrc.Dag
        for ctrl in [self.fk.ctrls[0], self.fk.ctrls[2]]:
            ctrl.ro = mrc.Ro.yxz

        self.ik.ctrl.ro = mrc.Ro.xzy
