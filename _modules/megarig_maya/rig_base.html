    """A base class to all rigging modules.
    A BaseRig object is initialized with a group as a Meta Node.
    Normally meta node is used as a group for controllers in the module.
    Meta data attributes are added to meta node so can be traced in the future.
        Meta data are module name, side and description.
    Default colors are also defined in this module.
        Left side -> red tone, right side -> blue tone, no side -> yellow tone.
    """

    def __init__(self, mod_name: str, mod_desc: str, side: mrc.Side):
        self.meta: mrc.Dag = mrc.Null()
        self.mod_name: str = mod_name
        self.mod_desc: str = mod_desc
        self.side: mrc.Side = side
        self.color: mrc.Color

        self.meta.name = mrnt.compose(side, (self.name, "rig"), "grp")

        metaattr = self.meta.add(ln=META_ATTR_NAME, dt="string")
        metaattr.lock = True

        self._add_meta_attrs(self.meta)

        # Colors
        self.color = mrc.Color.yellow
        self.soft_color = mrc.Color.soft_yellow
        self.dark_color = mrc.Color.dark_yellow
        if self.side == mrc.Side.left:
            self.color = mrc.Color.red
            self.soft_color = mrc.Color.soft_red
            self.dark_color = mrc.Color.dark_red
        elif self.side == mrc.Side.right:
            self.color = mrc.Color.blue
            self.soft_color = mrc.Color.soft_blue
            self.dark_color = mrc.Color.dark_blue

    @property
    def name(self) -> str:
        """Returns joined module name and module description"""
        return "_".join([n for n in [self.mod_name, self.mod_desc] if n])


[docs]    def current_names(self, names: Tuple[str]) -> Tuple[str]:
        """Inserts module name and description to the givne names

        Ex.
        module name = "torso"
        description = "a"
        names = ["spine", "1"]
        Returns: ("torso", "a", "spine", "1")

        So when it's composed -> "torso_a_spine_1_type"
        """
        return tuple([self.name] + list(names))


    def _add_meta_attrs(self, node: Union[mrc.Node, mrc.Dag]):
        if not node.attr(SIDE_ATTR_NAME).exists:
            sideattr = node.add(ln=SIDE_ATTR_NAME, dt="string")
            sideattr.v = self.side
            sideattr.lock = True

        if not node.attr(MOD_ATTR_NAME).exists:
            modattr = node.add(ln=MOD_ATTR_NAME, dt="string")
            modattr.v = self.mod_name
            modattr.lock = True

        if not node.attr(DESC_ATTR_NAME):
            descattr = node.add(ln=DESC_ATTR_NAME, dt="string")
            descattr.v = self.mod_desc
            descattr.lock = True


[docs]    def init_node(
        self,
        node: mrc.Node,
        side: mrc.Side,
        names: Tuple[str],
        type_: str,
    ) -> mrc.Node:
        """Initializes the given node to a rig node in this module.
        Initializing renames and adds meta attributes to the given node.

        Ex.
        module name = "arm"
        description = ""
        init_node(
            mrc.create("addDoubleLinear"), mrc.Side.left, ("arm", "1"), "add"
        ) -> l_arm_1_add
        """
        node.name = mrnt.compose(side, self.current_names(names), type_)
        self._add_meta_attrs(node)

        return node

[docs]    def add_controller(
        self,
        side: mrc.Side,
        names: Tuple[str],
        shape: mrc.Cp,
        color: mrc.Color,
        locked_attrs: Tuple[str],
    ) -> Tuple[mrc.Dag, mrc.Dag, mrc.Dag, mrc.Dag]:
        """Adds a controller with its groups

        Returns:
            mrc.Dag: Controller
            mrc.Dag: Offset group
            mrc.Dag: Orient group
            mrc.Dag: Zero group
        """
        ctrl = self.init_dag(mrc.Controller(shape), side, names, "ctrl")
        ctrl.lhattrs(*locked_attrs)
        if color:
            ctrl.color = color

        names = names + ("ctrl",)
        ofst = self.init_dag(mrc.group(ctrl), side, names, "ofst")
        ori = self.init_dag(mrc.group(ofst), side, names, "ori")
        zr = self.init_dag(mrc.group(ori), side, names, "zr")

        return ctrl, ofst, ori, zr



[docs]    def add_ik_controller(
        self,
        side: mrc.Side,
        names: Tuple[str],
        shape: mrc.Cp,
        color: mrc.Color,
        locked_attrs: List[str],
        target: mrc.Dag,
    ) -> Tuple[mrc.Dag, mrc.Dag, mrc.Dag, mrc.Dag]:
        """IK controller is a JointController if the target is given,
        zero group position is set at the target and the controller
        is oriented to the target.

        Returns:
            mrc.Dag: Controller
            mrc.Dag: Offset group
            mrc.Dag: Orient group
            mrc.Dag: Zero group
        """
        ctrl = self.init_dag(mrc.JointController(shape), side, names, "ctrl")
        if color:
            ctrl.color = color

        names = names + ("ctrl",)
        ofst = self.init_dag(mrc.group(ctrl), side, names, "ofst")
        ori = self.init_dag(mrc.group(ofst), side, names, "ori")
        zr = self.init_dag(mrc.group(ori), side, names, "zr")

        if target:
            zr.snap_point(target)
            ctrl.snap_orient(target)
            ctrl.freeze(r=True, s=False, t=False)

        ctrl.lhattrs(*locked_attrs)

        for grp in (ofst, ori, zr):
            grp.lhattrs("s", "v")

        return ctrl, ofst, ori, zr



[docs]    def add_curve_guide(
        self,
        root: mrc.Dag,
        end: mrc.Dag,
        side: mrc.Side,
        names: Tuple[str],
    ) -> Tuple[mrc.Dag, mrc.Node, mrc.Dag, mrc.Node, mrc.Dag]:
        """Creates a NURBs curve guide from the root to the end."""
        (
            curve,
            root_cluster,
            root_cluster_handle,
            end_cluster,
            end_cluster_handle,
        ) = mru.curve_guide(root, end)

        curve = self.init_dag(curve, side, names, "crv")
        root_cluster = self.init_node(root_cluster, side, names, "cluster")
        end_cluster = self.init_node(
            end_cluster, side, names + ("end",), "cluster"
        )

        return (
            curve,
            root_cluster,
            root_cluster_handle,
            end_cluster,
            end_cluster_handle,
        )



[docs]    def add_space_blender(
        self,
        target_nodes_and_attr_names: Tuple[Tuple[mrc.Dag, str]],
        cons_func: Union[mrc.pntcons, mrc.oricons, mrc.parcons],
        constrained_node: mrc.Dag,
        controller: mrc.Dag,
    ) -> Tuple[List[mrc.Dag], List[mrc.Dag], List[mrc.Node], List[str]]:
        """Adds space blender.
        Refer to pru.add_space_blender for the usage.

        Args:
            target_nodes_and_attr_names (tuple):
            cons_func (mrc.orient_constraint/mrc.point_constraint/
                mrc.parent_constraint):
            constrained_node (mrc.Dag):
            controller (mrc.Dag):
            pivot (mrc.Dag):
        """
        spc_nodes, spc_conds, attrs = mru.add_space_blender(
            target_nodes_and_attr_names,
            cons_func,
            constrained_node,
            controller,
        )

        side, names, type_ = mrnt.tokenize(controller.name)
        for spcnode, spccond, attr in zip(spc_nodes, spc_conds, attrs):
            curr_name = "_".join(names).replace(self.name, "")
            self.init_dag(spcnode, side, (curr_name, type_, attr, "tg"), "grp")
            self.init_node(
                spccond, side, (curr_name, type_, attr, "tg"), "cond"
            )

        return spc_nodes, spc_conds, attrs

[docs]    def add_attr_adder_amper(
        self,
        driver: mrc.Attr,
        driven: mrc.Attr,
        mult: float,
        names: Tuple[str],
    ) -> Tuple[mrc.Node, mrc.Node]:
        """Adds attribute adder with amper from the driver to the driven.
        This is normally used in Stretchy FK with Offset Translation.

        Returns:
            mrc.Node: addDoubleLinear
            mrc.Node: multDoubleLinear
        """
        add, amp = mru.attr_adder_amper(driver, driven, mult)

        add = self.init_node(add, self.side, names, "add")
        amp = self.init_node(amp, self.side, names, "amp")

        return add, amp



[docs]    def add_non_roll_joint(
        self,
        dag: mrc.Dag,
        axis: str,
        names: Tuple[str],
    ):
        """
        Args:
            dag: The target DAG object.
            axis: The twist(roll) axis valid strs are x, y and z
            names: A tuple of strings, is used to naming nodes.
        """
        nrjnt, pntcons, oricons, e2q, q2etw, q2esw, qinv, qprod = (
            mru.add_non_roll_joint(dag, axis, dag.get_parent())
        )

        nrjnt = self.init_dag(nrjnt, self.side, names, "nrJnt")
        e2q = self.init_node(e2q, self.side, names + ("nr",), "e2q")
        q2etw = self.init_node(q2etw, self.side, names + ("nr", "tw"), "q2e")
        q2esw = self.init_node(q2esw, self.side, names + ("nr", "sw"), "q2e")
        qinv = self.init_node(qinv, self.side, names + ("nr",), "qinv")
        qprod = self.init_node(qprod, self.side, names + ("nr",), "qprod")

        pntcons.name = nrjnt.name + "_pointConstraint"
        oricons.name = nrjnt.name + "_orientConstraint"

        return nrjnt, e2q, q2etw, q2esw, qinv, qprod
