[docs]def set_default_finger_multipliers(ctrl: mrc.Dag, finger: str):
    """Sets default values to finger multipliers regarding to FINGER_SDK."""
    value_dict = None

    for key in FINGER_SDK.keys():
        if finger.startswith(key):
            value_dict = FINGER_SDK[key]

    if value_dict == None:
        return False

    value_dict: dict
    for attr in value_dict.keys():
        attrname = "{f}{a}".format(f=finger, a=attr)
        if ctrl.attr(attrname).exists:
            ctrl.attr(attrname).v = value_dict[attr]

    return True

        guides: List[str],

        vol_pres: bool,
    ):
        super(FingerRig, self).__init__(
            guides=guides[:-1],
            side=side,
            mod_name=finger,
            mod_desc="",
            joint_type="jnt",
            ctrl_shape=mrc.Cp.cube,
            color=mrc.Color.none,
            parent=parent,
            spaces_attrnames=None,
        )

        # Hand Ctrl
        self.hand_ctrl = hand_ctrl
        self.vol_pres_vis_attr = self.hand_ctrl.get_shape().attr("volPresVis")

        # Tip Joint
        self.tip_guide = mrc.Dag(guides[-1])
        self.tip_jnt = self.init_dag(
            mru.create_joint_at(self.tip_guide), self.side, ("tip",), "jnt"
        )
        self.tip_jnt.set_parent(self.jnts[-1])
        if skin_parent:
            self.jnts[0].set_parent(skin_parent)

        # Color
        self.ctrls[0].color = self.color
        self.ctrls[1].color = self.dark_color
        for ctrl in self.ctrls[2:]:
            ctrl.color = self.soft_color

        # Stretch Attr
        str_attr = self.ctrls[-1].add(ln="stretch", k=True)
        str_dv = self.tip_jnt.attr("ty").v
        self.tip_str_amper = self.init_node(
            mru.attr_amper(str_attr, self.tip_jnt.attr("ty"), 0.1),
            self.side,
            ("tip", "str"),
            "amp",
        )
        self.tip_str_adder = self.init_node(
            mru.attr_adder(
                self.tip_str_amper.attr("o"), self.tip_jnt.attr("ty"), str_dv
            ),
            self.side,
            ("tip", "str"),
            "add",
        )

        # Volume Preservation
        self.vol_pres_rigs: List[rig_volpres.VolPresRig] = []
        if vol_pres:
            volpres_ofst = mru.get_length(self.jnts + [self.tip_jnt]) * 0.05
            for ix, jnt in enumerate(self.jnts[1:]):
                modname = "_".join(
                    [n for n in [finger, str(ix + 2), "volpres"] if n]
                )
                volpres = rig_volpres.VolPresRig(
                    target=jnt,
                    target_pos=jnt,
                    parent=self.meta,
                    offset=volpres_ofst,
                    volume_joints=(False, False, False, False, True, True),
                    mod_name=modname,
                    desc="",
                    side=self.side,
                    joint_type="jnt",
                )
                self.vol_pres_vis_attr >> volpres.meta.attr("v")
                self.vol_pres_rigs.append(volpres)

        # Rotation Sum Nodes
        self.sdk_rx_sums: List[mrc.Node] = []
        self.sdk_ry_sums: List[mrc.Node] = []
        self.sdk_rz_sums: List[mrc.Node] = []

        for ix in range(len(self.jnts)):
            idstr = str(ix + 1)
            rx_sum = self.init_node(
                mrc.create("plusMinusAverage"),
                self.side,
                ("rx", "sdk", idstr),
                "sum",

            self.sdk_ry_sums.append(ry_sum)
            self.sdk_rz_sums.append(rz_sum)

            rx_sum.attr("o1") >> self.ctrl_ories[ix].attr("rx")
            ry_sum.attr("o1") >> self.ctrl_ories[ix].attr("ry")
            rz_sum.attr("o1") >> self.ctrl_ories[ix].attr("rz")

        # Hand SDKs
        if not self.hand_ctrl.attr("__hand__").exists:
            mru.add_divide_attr(self.hand_ctrl, "hand")

        # Hand SDKs - fist
        if not self.hand_ctrl.attr("fist").exists:
            self.hand_ctrl.add(ln="fist", k=True)

        self.fist_ry_ampers: List[mrc.Node] = []
        self.fist_rz_ampers: List[mrc.Node] = []

        # Skin Joints
        self.add_nodes_to_set(self.jnts + [self.tip_jnt], mrc.SKIN_SET)

        # RO
        ctrl: mrc.Dag
        for ctrl in self.ctrls:
            ctrl.ro = mrc.Ro.yzx

    def _hand_sdk(
        self,
        sdk_name: str,
        attr_name: str,
        index: str,
        sum_node: mrc.Node,
        dv: float,
    ) -> mrc.Node:
        amp_attr = self.hand_ctrl.get_shape().add(
            ln=mrnt.lower_camelize(
                [self.mod_name, self.mod_desc, sdk_name, attr_name, index]
            ),
            dv=dv,
            k=True,
        )

        conns = sum_node.attr("i1").connections(s=True, d=False)
        last_input = 0
        if conns:
            last_input = len(conns)

        amper = mru.attr_amper(
            self.hand_ctrl.attr(sdk_name),
            sum_node.attr("i1[{}]".format(last_input)),
            dv,
        )
        amp_attr >> amper.attr("i2")

        return self.init_node(
            amper, self.side, (sdk_name, attr_name, index), "mult"
        )

    def __init__(
        self,
        thumb_guides: List[str],
        index_guides: List[str],
        middle_guides: List[str],
        ring_guides: List[str],
        pinky_guides: List[str],
        mod_desc: str,
        side: mrc.Side,
        parent: mrc.Dag,
        skin_parent: mrc.Dag,
        vol_pres: bool,
    ):
        super(HumanHand, self).__init__("hand", mod_desc, side)
        if parent:
            self.meta.set_parent_at(parent)

        # Main Controllers
        (
            self.ctrl,
            self.ctrl_ofst,
            self.ctrl_ori,
            self.ctrl_zr,
        ) = self.add_controller(
            self.side, ("",), mrc.Cp.stick, self.color, ("s", "v")
        )
        self.ctrl_zr.set_parent_at(self.meta)
        self.main_shape = self.ctrl.get_shape()

        if vol_pres:
            self.vol_pres_vis_attr = self.ctrl.get_shape().add(
                ln="volPresVis", min=0, max=1, k=True, dv=0
            )

        self.thumb = None
        self.index = None
        self.middle = None
        self.ring = None
        self.pinky = None

        if thumb_guides:
            self.thumb = FingerRig(
                guides=thumb_guides,
                hand_ctrl=self.ctrl,
                finger="thumb",
                side=self.side,
                parent=self.meta,
                skin_parent=skin_parent,
                vol_pres=vol_pres,
            )
        if index_guides:
            self.index = FingerRig(
                guides=index_guides,
                hand_ctrl=self.ctrl,
                finger="index",
                side=self.side,
                parent=self.meta,
                skin_parent=skin_parent,
                vol_pres=vol_pres,
            )
        if middle_guides:
            self.middle = FingerRig(
                guides=middle_guides,
                hand_ctrl=self.ctrl,
                finger="middle",
                side=self.side,
                parent=self.meta,
                skin_parent=skin_parent,
                vol_pres=vol_pres,
            )
        if ring_guides:
            self.ring = FingerRig(
                guides=ring_guides,
                hand_ctrl=self.ctrl,
                finger="ring",
                side=self.side,
                parent=self.meta,
                skin_parent=skin_parent,
                vol_pres=vol_pres,
            )
        if pinky_guides:
            self.pinky = FingerRig(
                guides=pinky_guides,
                hand_ctrl=self.ctrl,
                finger="pinky",
                side=self.side,
                parent=self.meta,
                skin_parent=skin_parent,
                vol_pres=vol_pres,
            )

        # Finger Constraints
        fingers: List[FingerRig] = []
        for finger in (
            self.thumb,
            self.index,
            self.middle,
            self.ring,
            self.pinky,
        ):
            if finger:
                fingers.append(finger)

        mid_cons = mrc.parcons(
            self.ctrl, self.ctrl_ori, fingers[2].ctrl_zrs[0], mo=True
        )

        if len(fingers) == 5:
            second_last_cons = mrc.parcons(
                self.ctrl, self.ctrl_ori, fingers[-2].ctrl_zrs[0], mo=True
            )
            mrc.parcons(self.ctrl, fingers[-1].ctrl_zrs[0], mo=True)

            mid_cons.attr("w0").v = 0.33333
            mid_cons.attr("w1").v = 0.66667

            second_last_cons.attr("w0").v = 0.66667
            second_last_cons.attr("w1").v = 0.33333

        elif len(fingers) == 4:
            last_cons = mrc.parcons(
                self.ctrl, self.ctrl_ori, fingers[-1].ctrl_zrs[0], mo=True
            )
            mid_cons.attr("w0").v = 0.2
            mid_cons.attr("w1").v = 0.8

            last_cons.attr("w0").v = 0.5
            last_cons.attr("w1").v = 0.5
