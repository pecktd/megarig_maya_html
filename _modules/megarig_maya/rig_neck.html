[docs]class NeckRig(rig_base.BaseRig):
    def __init__(
        self,
        guide: str,
        tip_guide: str,
        joint_amount: int,
        mod_desc: str,
        spaces_attrnames: Union[Tuple[Tuple[mrc.Dag, str]], None],
        parent: mrc.Dag,
        skin_parent: mrc.Dag,
        still_parent: mrc.Dag,
    ):
        # Main Groups
        super(NeckRig, self).__init__("neck", mod_desc, mrc.Side.none)
        self.meta.set_parent(parent)

        self.still_grp = self.init_dag(
            mrc.Null(), self.side, ("still",), "grp"
        )
        self.still_grp.set_parent(still_parent)

        self.guide = mrc.Dag(guide)
        self.tip_guide = mrc.Dag(tip_guide)

        # Joints
        self.jnt = self.init_dag(
            mru.create_joint_at(self.guide), self.side, ("",), "jnt"
        )
        self.tip_jnt = self.init_dag(
            mru.create_joint_at(self.guide), self.side, ("tip",), "jnt"
        )
        self.tip_jnt.snap_point(self.tip_guide)
        self.tip_jnt.set_parent(self.jnt)
        self.jnt.set_parent(skin_parent)

        # IK Handle
        self.ikh = self.init_dag(
            mrc.ikrp(self.jnt, self.tip_jnt), self.side, ("",), "ikh"
        )

        # Main Controller
        (
            self.ctrl,
            self.ctrl_ofst,
            self.ctrl_ori,
            self.ctrl_zr,
        ) = self.add_controller(
            self.side, ("",), mrc.Cp.double_circle, self.color, ("s", "v")
        )
        self.ctrl_zr.snap(self.guide)
        self.ctrl_zr.set_parent(self.meta)
        mrc.pntcons(self.ctrl, self.jnt)

        # Head Space
        self.head_spc = self.init_dag(
            mrc.Null(), self.side, ("head", "spc"), "grp"
        )
        self.head_spc.snap(self.tip_jnt)
        self.head_spc.set_parent(self.ctrl)
        self.ikh.set_parent(self.head_spc)
        self.ikh.hide = True
        self.head_nrj, _, _, _, _, _ = self.add_non_roll_joint(
            self.head_spc, "y", ("head",)
        )
        self.head_nrj.hide = True

        # Stretchy IK
        self.str_attr = self.ctrl.add(ln="stretch", k=True)

        self.root_pos = self.init_dag(mrc.Null(), self.side, ("root",), "pos")
        self.end_pos = self.init_dag(mrc.Null(), self.side, ("end",), "pos")

        self.root_pos.set_parent(self.meta)
        self.end_pos.set_parent(self.meta)

        mrc.pntcons(self.ctrl, self.root_pos)
        mrc.pntcons(self.head_spc, self.end_pos)

        self.dist = self.init_node(
            mrc.create("distanceBetween"), self.side, ("",), "dist"
        )
        self.root_pos.attr("t") >> self.dist.attr("p1")
        self.end_pos.attr("t") >> self.dist.attr("p2")
        self.orig_len = self.dist.attr("d").v

        self.at_str_div = self.init_node(
            mrc.create("multiplyDivide"),
            self.side,
            (
                "at",
                "str",
            ),
            "div",
        )
        self.at_str_div.attr("op").v = mrc.Op.divide
        self.dist.attr("d") >> self.at_str_div.attr("i1x")
        self.at_str_div.attr("i2x").v = self.orig_len

        self.at_str_mult = self.init_node(
            mrc.create("multDoubleLinear"),
            self.side,
            (
                "at",
                "str",
            ),
            "mult",
        )
        self.at_str_div.attr("ox") >> self.at_str_mult.attr("i1")
        self.at_str_mult.attr("i2").v = self.orig_len

        self.at_str_cond = self.init_node(
            mrc.create("condition"),
            self.side,
            (
                "at",
                "str",
            ),
            "cond",
        )
        self.at_str_cond.attr("op").v = mrc.Op.greater_than
        self.dist.attr("d") >> self.at_str_cond.attr("ft")
        self.at_str_cond.attr("st").v = self.orig_len
        self.at_str_cond.attr("cfr").v = self.orig_len
        self.at_str_mult.attr("o") >> self.at_str_cond.attr("ctr")

        self.at_str_blend = self.init_node(
            mrc.create("blendTwoAttr"), self.side, ("at", "str"), "blend"
        )
        self.auto_stretch_attr = self.at_str_blend.attr("ab")
        self.at_str_blend.attr("i[0]").v = self.orig_len
        self.at_str_cond.attr("ocr") >> self.at_str_blend.attr("i[1]")

        self.str_mult = self.init_node(
            mrc.create("multDoubleLinear"), self.side, ("str",), "mult"
        )
        self.str_mult.attr("i2").v = 0.1
        self.str_attr >> self.str_mult.attr("i1")

        self.str_add = self.init_node(
            mrc.create("addDoubleLinear"), self.side, ("str"), "add"
        )
        self.str_mult.attr("o") >> self.str_add.attr("i1")
        self.at_str_blend.attr("o") >> self.str_add.attr("i2")
        self.str_add.attr("o") >> self.tip_jnt.attr("ty")

        # Ribbon IK
        if joint_amount:
            self.ribik = rig_ribbon_ik.RibbonIk(
                root=self.jnt,
                end=self.head_nrj,
                root_tg=self.jnt,
                end_tg=self.tip_jnt,
                parent_tg=None,
                jnt_amt=4,
                side=self.side,
                mod_name="ik_neck",
                desc="",
                ctrl_shp=mrc.Cp.double_circle,
                dtl_ctrl_shp=mrc.Cp.none,
                ctrl_clr=self.dark_color,
                dtl_clr=self.color,
                parent=self.ctrl,
                skin_parent=self.jnt,
                still_parent=self.still_grp,
            )
            self.head_nrj.attr("twist") >> self.ribik.end_twist

        if spaces_attrnames:
            self.add_space_blender(
                spaces_attrnames,
                mrc.oricons,
                self.ctrl_zr,
                self.ctrl,
            )

        self.add_nodes_to_set(self.ribik.ribbon.jnts, mrc.SKIN_SET)

        # RO
        self.ctrl.ro = mrc.Ro.xzy
