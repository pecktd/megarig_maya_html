[docs]class StretchyIk(rig_base.BaseRig):
    def __init__(
        self,
        guides: Tuple[str],
        ee_index: int,
        pole_vector_offset: float,
        actual_length: bool,
        spaces_attrnames: Union[tuple, None],
        mod_name: str,
        desc: str,
        ctrl_shape: mrc.Cp,
        ctrl_color: mrc.Color,
        joint_type: str,
        side: mrc.Side,
        parent: mrc.Dag,
    ):
        """Creates Stretchy IK joint chain"""
        super(StretchyIk, self).__init__(mod_name, desc, side)

        self.guides = mrc.to_dags(guides)
        self.ee_index = ee_index

        if parent:
            self.meta.snap(parent)
            self.meta.set_parent(parent)

        # IK Joints
        def _create_jnts() -> List[mrc.Dag]:
            jnts = []
            for ix, guide in enumerate(self.guides):
                curr_jnt = self.init_dag(
                    mru.create_joint_at(guide),
                    self.side,
                    (str(ix + 1),),
                    joint_type,
                )
                if ix > 0:
                    curr_jnt.set_parent(jnts[-1])
                else:
                    curr_jnt.attr("ssc").v = False
                    curr_jnt.set_parent(self.meta)
                jnts.append(curr_jnt)

            return jnts

        self.jnts = _create_jnts()

        # Distance Attributes
        mru.add_divide_attr(self.meta, "distances")

        # Original distance from root to IKH
        chain_len = om.MVector(
            om.MVector(*self.jnts[0].world_pos)
            - om.MVector(*self.jnts[self.ee_index].world_pos)
        ).length()
        self.orig_ikh_dist_attr = self.meta.add(
            ln="origIkhDist", dv=chain_len, k=True
        )

        # Sum of all joint lengthes
        self.skel_len_attr = self.meta.add(ln="skelLen", k=True)

        # Creates IK Joint chain at the given template joints
        skel_lens = []
        bone_len = 0.0
        for ix in range(len(self.jnts[: self.ee_index + 1])):
            if ix == 0:
                continue
            prev_skel_len = self.jnts[ix].attr("ty").v
            if actual_length:
                prev_skel_len = om.MVector(
                    om.MVector(*self.jnts[ix].world_pos)
                    - om.MVector(*self.jnts[ix - 1].world_pos)
                ).length()
                if side == mrc.Side.right:
                    prev_skel_len *= -1
                bone_len += abs(prev_skel_len)

            skel_lens.append(prev_skel_len)

        self.skel_len_attr.v = self.meta.attr("origIkhDist").v
        if actual_length:
            self.skel_len_attr.v = bone_len
        self.skel_len_attr.lock = True

        # IK Ctrls
        (
            self.root_ctrl,
            self.root_ctrl_ori,
            self.root_ctrl_ofst,
            self.root_ctrl_zr,
        ) = self.add_ik_controller(
            self.side,
            ("root",),
            ctrl_shape,
            ctrl_color,
            ("r", "s", "v", "ro"),
            self.jnts[0],
        )
        self.root_ctrl_zr.set_parent(self.meta)

        (
            self.ctrl,
            self.ctrl_ofst,
            self.ctrl_ori,
            self.ctrl_zr,
        ) = self.add_ik_controller(
            self.side,
            ("",),
            ctrl_shape,
            ctrl_color,
            ("s", "v"),
            self.jnts[self.ee_index],
        )
        self.ctrl_zr.set_parent(self.meta)

        if spaces_attrnames:
            self.add_space_blender(
                spaces_attrnames,
                mrc.parcons,
                self.ctrl_zr,
                self.ctrl,
            )

        # Twist
        self.twist_attr = self.ctrl.add(ln="twist", k=True)

        # Soft IK Attr
        if actual_length:
            self.soft_ik_attr = self.ctrl.add(
                ln="softIk", k=True, min=0, max=1, dv=1
            )

        # Stretchy IK
        mru.add_divide_attr(self.ctrl, "stretch")
        self.auto_stretch_attr = self.ctrl.add(
            ln="autoStretch", k=True, min=0, max=1, dv=1
        )

        # IK Pos
        self.ikh_pos = self.init_dag(mrc.Null(), self.side, ("pos",), "grp")
        self.ikh_pos_zr = self.init_dag(
            mrc.group(self.ikh_pos), self.side, ("pos",), "zr"
        )
        self.ikh_pos_zr.snap(self.jnts[self.ee_index])
        self.ikh_pos_zr.set_parent(self.ctrl)

        # IK Root Pos
        self.root_pos = self.init_dag(
            mrc.Null(), self.side, ("root", "pos"), "grp"
        )
        self.root_zr = self.init_dag(
            mrc.group(self.root_pos), self.side, ("root", "pos"), "zr"
        )
        self.root_zr.snap(self.jnts[0])
        self.root_zr.set_parent(self.root_ctrl)

        mrc.point_constraint(self.root_pos, self.jnts[0])

        # Finds position of pole vector
        st_pos = om.MVector(*self.jnts[0].world_pos)
        ee_pos = om.MVector(*self.jnts[self.ee_index].world_pos)

        # IK chain vector
        chain_vec = om.MVector(ee_pos - st_pos)
        chain_len = chain_vec.length()
        chain_vec.normalize()

        # The length of the joint to the projection to the chain vector
        proj_len = 0.0
        pole_vctr_pos = om.MVector()
        for ikjnt in self.jnts[1 : self.ee_index + 1]:
            currjnt_pos = om.MVector(*ikjnt.world_pos)
            currjnt_vec = currjnt_pos - st_pos

            # A vector from projection point to current joint.
            proj_dot = currjnt_vec * chain_vec
            proj_vctr = om.MVector(currjnt_vec - chain_vec * proj_dot)
            if proj_vctr.length() > proj_len:
                # If current projection has the longest distance.
                proj_len = proj_vctr.length()
                proj_vctr.normalize()
                pole_vctr_pos = (
                    st_pos
                    + (chain_vec * proj_dot)
                    + (proj_vctr * chain_len * pole_vector_offset)
                )

        # Pole Vector Controller
        self.pole_vec_ctrl = self.init_dag(
            mrc.Controller(mrc.Cp.plus), self.side, ("pole", "vec"), "ctrl"
        )
        self.pole_vec_ctrl.color = ctrl_color
        self.pole_vec_ctrl.lhattrs("r", "s", "v")
        self.pole_vec_ctrl_ofst = self.init_dag(
            mrc.group(self.pole_vec_ctrl),
            self.side,
            ("pole", "vec", "ctrl"),
            "ofst",
        )
        self.pole_vec_ctrl_ori = self.init_dag(
            mrc.group(self.pole_vec_ctrl_ofst),
            self.side,
            ("pole", "vec", "ctrl"),
            "ori",
        )
        self.pole_vec_ctrl_zr = self.init_dag(
            mrc.group(self.pole_vec_ctrl_ori),
            self.side,
            ("pole", "vec", "ctrl"),
            "zr",
        )
        self.pole_vec_ctrl_zr.world_pos = (
            pole_vctr_pos.x,
            pole_vctr_pos.y,
            pole_vctr_pos.z,
        )
        self.pole_vec_ctrl_zr.set_parent(self.meta)

        if spaces_attrnames:
            spcs, _, _ = self.add_space_blender(
                spaces_attrnames,
                mrc.parent_constraint,
                self.pole_vec_ctrl_ori,
                self.pole_vec_ctrl,
            )
            mrc.parcons(self.ctrl, spcs[0], mo=True)

        # Pole Vector Guides
        def _add_curve_guide(jnt):
            (
                crv,
                _,
                _,
                _,
                _,
            ) = self.add_curve_guide(
                jnt,
                self.pole_vec_ctrl,
                self.side,
                (
                    "pole",
                    "vec",
                    "guide",
                ),
            )
            crv.set_parent(self.meta)
            crv.attr("t").v = (0,) * 3
            crv.attr("r").v = (0,) * 3

            return crv

        self.pole_crv = _add_curve_guide(self.jnts[1])

        # IK Handle
        self.ikh = self.init_dag(
            mrc.ikrp(sj=self.jnts[0], ee=self.jnts[self.ee_index]),
            self.side,
            ("",),
            "ikh",
        )
        mc.poleVectorConstraint(self.pole_vec_ctrl, self.ikh)
        self.ikh.set_parent(self.meta)
        self.ikh_cons = mrc.point_constraint(self.ikh_pos, self.ikh)
        self.ikh.hide = True

        self.ctrl.attr("twist") >> self.ikh.attr("twist")

        # IK Distance
        self.root_dist_pos = self.init_dag(
            mrc.Null(), self.side, ("root", "dist"), "pos"
        )
        self.ikh_dist_pos = self.init_dag(
            mrc.Null(), self.side, ("ik", "dist"), "pos"
        )

        mrc.pntcons(self.root_pos, self.root_dist_pos)
        mrc.pntcons(self.ikh_pos, self.ikh_dist_pos)

        self.root_dist_pos.set_parent(self.meta)
        self.ikh_dist_pos.set_parent(self.meta)

        self.ikh_dist = self.init_node(
            mrc.create("distanceBetween"), self.side, ("ikh",), "dist"
        )
        self.root_dist_pos.attr("t") >> self.ikh_dist.attr("p1")
        self.ikh_dist_pos.attr("t") >> self.ikh_dist.attr("p2")

        self.ikh_len = self.ikh_dist.attr("d")

        # Stretchy IK
        mru.add_divide_attr(self.meta, "boneLength")
        self.str_amps: List[mrc.Node] = []
        self.str_mults: List[mrc.Node] = []
        self.str_adds: List[mrc.Node] = []
        orig_len_attrs: List[mrc.Attr] = []
        for ix in range(len(self.jnts[: self.ee_index])):
            idstr = str(ix + 1)
            orig_len_attr = self.meta.add(
                ln="origLen{}".format(idstr),
                k=True,
                dv=skel_lens[ix],
            )
            orig_len_attrs.append(orig_len_attr)
            orig_len_attr.lock = True

            str_attr = self.ctrl.add(ln="stretch{}".format(idstr), k=True)
            str_amp_attr = self.meta.add(
                ln="strAmp{}".format(idstr), k=True, min=0, max=1, dv=0.1
            )

            str_amp = self.init_node(
                mrc.create("multDoubleLinear"),
                self.side,
                ("str", idstr),
                "amp",
            )
            self.str_amps.append(str_amp)
            str_amp_attr >> str_amp.attr("i1")
            orig_len_attr >> str_amp.attr("i2")

            str_mult = self.init_node(
                mrc.create("multDoubleLinear"),
                self.side,
                ("str", idstr),
                "mult",
            )
            self.str_mults.append(str_mult)
            str_amp.attr("o") >> str_mult.attr("i1")
            str_attr >> str_mult.attr("i2")

            str_add = self.init_node(
                mrc.create("addDoubleLinear"), self.side, ("str", idstr), "add"
            )
            self.str_adds.append(str_add)
            str_add.attr("i1").v = orig_len_attr.v
            str_mult.attr("o") >> str_add.attr("i2")
            str_add.attr("o") >> self.jnts[ix + 1].attr("ty")

        # Auto-Stretch IK
        self.atstr_cnd = self.init_node(
            mrc.create("condition"), self.side, ("at", "str"), "cond"
        )
        self.atstr_cnd.attr("op").v = mrc.Op.greater_than
        self.atstr_cnd.attr("cfr").v = 1
        self.ikh_len >> self.atstr_cnd.attr("ft")
        self.skel_len_attr >> self.atstr_cnd.attr("st")

        self.atstr_div = self.init_node(
            mrc.create("multiplyDivide"), self.side, ("at", "str"), "div"
        )
        self.atstr_div.attr("op").v = mrc.Op.divide
        self.ikh_len >> self.atstr_div.attr("i1x")
        self.skel_len_attr >> self.atstr_div.attr("i2x")
        self.atstr_div.attr("ox") >> self.atstr_cnd.attr("ctr")

        self.atstr_mults = []
        for ix in range(len(self.jnts[: self.ee_index])):
            idstr = str(ix + 1)
            curr_origlen_attr = orig_len_attrs[ix]
            atstr_mult = self.init_node(
                mrc.create("multDoubleLinear"),
                self.side,
                ("at", "str", idstr),
                "mult",
            )
            curr_origlen_attr >> atstr_mult.attr("i1")
            self.atstr_cnd.attr("ocr") >> atstr_mult.attr("i2")

            atstr_blend = self.init_node(
                mrc.create("blendTwoAttr"),
                self.side,
                ("at", "str", idstr),
                "blend",
            )
            self.ctrl.attr("autoStretch") >> atstr_blend.attr("ab")
            curr_origlen_attr >> atstr_blend.attr("i[0]")
            atstr_mult.attr("o") >> atstr_blend.attr("i[1]")

            atstr_blend.attr("o") >> self.str_adds[ix].attr("i1")

        # Adds Soft IK if the actual_length is True.
        if actual_length:
            self._add_soft_ik()

    def _add_soft_ik(self):
        """Adds Soft IK calculations if the actual_length is True."""
        # IK Vector Chain
        # A single chain joint from root to ee
        self.vec_root_rj = self.init_dag(
            mru.create_joint_at(self.jnts[0]),
            self.side,
            ("vec", "root"),
            "rigJnt",
        )
        self.vec_tip_rj = self.init_dag(
            mru.create_joint_at(self.jnts[self.ee_index]),
            self.side,
            ("vec", "tip"),
            "rigJnt",
        )
        self.vec_tip_rj.set_parent(self.vec_root_rj)
        mru.orient_bone(self.vec_root_rj, (1, 0, 0), self.vec_root_rj.x_axis)
        self.vec_root_rj.set_parent(self.meta)
        mrc.point_constraint(self.root_pos, self.vec_root_rj)
        self.vec_root_rj.hide = True

        self.soft_pos = self.init_dag(
            mrc.Null(), self.side, ("soft", "pos"), "grp"
        )
        self.soft_pos.snap(self.vec_tip_rj)
        self.soft_pos.set_parent(self.vec_root_rj)
        self.curr_pos = self.init_dag(
            mrc.Null(), self.side, ("curr", "pos"), "grp"
        )
        self.curr_pos.snap(self.vec_tip_rj)
        self.curr_pos.set_parent(self.vec_root_rj)
        mrc.point_constraint(self.ikh_pos, self.curr_pos)

        self.vec_ikh = self.init_dag(
            mrc.ikrp(sj=self.vec_root_rj, ee=self.vec_tip_rj),
            self.side,
            ("vec",),
            "ikh",
        )
        self.vec_ikh.set_parent(self.meta)
        mrc.point_constraint(self.ikh_pos, self.vec_ikh)
        self.vec_ikh.hide = True

        self.orig_ikh_dist_attr.v = self.vec_tip_rj.attr("ty").v
        self.orig_ikh_dist_attr.lock = True

        # Soft distances.
        mru.add_divide_attr(self.meta, "softIk")
        self.sik_max_dist_attr = self.meta.add(ln="softIkMaxDist", k=True)
        self.sik_dist_attr = self.meta.add(ln="softIkDist", k=True)

        # e = 2.71828
        # Equation dSoft*(1 - pow(e, -(x - dA)/dSoft)) + dA

        # dSoft, Soft IK distance.
        # softIkMaxDist is a difference between a length of skeleton and
        # a length of IK distance.
        self.sik_max_dist_attr.v = (
            self.skel_len_attr.v - self.orig_ikh_dist_attr.v
        )

        five_of_ik_dist = self.orig_ikh_dist_attr.v * 0.05
        if self.sik_max_dist_attr.v < five_of_ik_dist:
            # If max soft distance less than 5 percent of IK distance.
            self.sik_dist_attr.v = self.sik_max_dist_attr.v
        else:
            self.sik_dist_attr.v = five_of_ik_dist

        self.sik_max_dist_attr.lock = True
        self.sik_dist_attr.lock = True

        self.dsoft_mult = self.init_node(
            mrc.create("multDoubleLinear"), self.side, ("d", "soft"), "mult"
        )
        self.sik_dist_attr >> self.dsoft_mult.attr("i1")
        self.soft_ik_attr >> self.dsoft_mult.attr("i2")

        # dChain, uses skelLen.
        # dA, dChain - dSoft
        self.da_sub = self.init_node(
            mrc.create("plusMinusAverage"), self.side, ("da",), "sub"
        )
        self.da_sub.attr("op").v = mrc.Op.subtract
        self.skel_len_attr >> self.da_sub.attr("i1[0]")
        self.dsoft_mult.attr("o") >> self.da_sub.attr("i1[1]")

        # -(x - dA), x is current distance.
        self.x_sub_da_sub = self.init_node(
            mrc.create("plusMinusAverage"), self.side, ("x", "da"), "sub"
        )
        self.x_sub_da_sub.attr("op").v = mrc.Op.subtract
        self.curr_pos.attr("ty") >> self.x_sub_da_sub.attr("i1[0]")
        self.da_sub.attr("o1") >> self.x_sub_da_sub.attr("i1[1]")

        self.x_sub_da_inv = self.init_node(
            mrc.create("multDoubleLinear"),
            self.side,
            ("x", "minus", "da"),
            "inv",
        )
        self.x_sub_da_sub.attr("o1") >> self.x_sub_da_inv.attr("i1")
        self.x_sub_da_inv.attr("i2").v = -1

        # -(x - dA)/dSoft), power of e.
        # dSoft has to be greater than 0.
        self.dsoft_no_zr = self.init_node(
            mrc.create("condition"), self.side, ("dsoft", "no", "zr"), "cond"
        )
        self.dsoft_no_zr.attr("op").v = mrc.Op.greater_than
        self.dsoft_no_zr.attr("st").v = 0
        self.dsoft_mult.attr("o") >> self.dsoft_no_zr.attr("ft")
        self.dsoft_mult.attr("o") >> self.dsoft_no_zr.attr("ctr")
        self.dsoft_no_zr.attr("cfr").v = 0.001

        # poe, power of e.
        self.poe_div = self.init_node(
            mrc.create("multiplyDivide"), self.side, ("poe",), "div"
        )
        self.poe_div.attr("op").v = mrc.Op.divide
        self.x_sub_da_inv.attr("o") >> self.poe_div.attr("i1x")
        self.dsoft_no_zr.attr("ocr") >> self.poe_div.attr("i2x")

        # power e, pow(e, -(x - dA)/dSoft).
        self.e_pow = self.init_node(
            mrc.create("multiplyDivide"), self.side, ("e",), "pow"
        )
        self.e_pow.attr("op").v = mrc.Op.power
        self.e_pow.attr("i1x").v = 2.71828
        self.poe_div.attr("ox") >> self.e_pow.attr("i2x")

        # 1 - pow(e, -(x - dA)/dSoft)
        self.e_pow_rev = self.init_node(
            mrc.create("reverse"), self.side, ("e", "pow"), "rev"
        )
        self.e_pow.attr("ox") >> self.e_pow_rev.attr("ix")

        # soft part, dSoft*(1 - pow(e, -(x - dA)/dSoft))
        self.soft_part_mult = self.init_node(
            mrc.create("multDoubleLinear"), self.side, ("soft", "part"), "mult"
        )
        self.dsoft_mult.attr("o") >> self.soft_part_mult.attr("i1")
        self.e_pow_rev.attr("ox") >> self.soft_part_mult.attr("i2")

        # soft distance.
        self.soft_dist_add = self.init_node(
            mrc.create("addDoubleLinear"), self.side, ("soft", "dist"), "add"
        )
        self.soft_part_mult.attr("o") >> self.soft_dist_add.attr("i1")
        self.da_sub.attr("o1") >> self.soft_dist_add.attr("i2")

        # Soft IK Condition - is activated when current distance > da
        self.soft_cnd = self.init_node(
            mrc.create("condition"), self.side, ("soft",), "cond"
        )
        self.soft_cnd.attr("op").v = mrc.Op.greater_or_equal
        self.curr_pos.attr("ty") >> self.soft_cnd.attr("ft")
        self.da_sub.attr("o1") >> self.soft_cnd.attr("st")
        self.soft_dist_add.attr("o") >> self.soft_cnd.attr("ctr")
        self.curr_pos.attr("ty") >> self.soft_cnd.attr("cfr")

        self.soft_at_str_mult = self.init_node(
            mrc.create("multDoubleLinear"),
            self.side,
            ("soft", "at", "str"),
            "mult",
        )
        self.atstr_cnd.attr("ocr") >> self.soft_at_str_mult.attr("i1")
        self.soft_cnd.attr("ocr") >> self.soft_at_str_mult.attr("i2")
        self.soft_at_str_mult.attr("o") >> self.soft_pos.attr("ty")
        mc.delete(self.ikh_cons)
        mrc.pntcons(self.soft_pos, self.ikh)
