    def __init__(
        self,
        mod_name: str,
        desc: str,
        side: mrc.Side,
        joint_type: str,
        offset: float,
    ):
        super(VolRig, self).__init__(mod_name, desc, side)

        self.pos = self.init_dag(mrc.Null(), self.side, ("pos",), "grp")

        self.pos_ofst = self.init_dag(
            mrc.group(self.pos), self.side, ("pos",), "ofst"
        )
        self.pos_sdk = self.init_dag(
            mrc.group(self.pos_ofst), self.side, ("pos",), "sdk"
        )
        self.pos_zr = self.init_dag(
            mrc.group(self.pos_sdk), self.side, ("pos",), "zr"
        )
        self.pos_zr.set_parent(self.meta)

        self.jnt = self.init_dag(mrc.Joint(), self.side, ("",), joint_type)

        # Sum Nodes
        self.t_sum = self.init_node(
            mrc.create("plusMinusAverage"), self.side, ("t",), "sum"
        )
        self.t_sum.attr("o3") >> self.pos_sdk.attr("t")

        self.r_sum = self.init_node(
            mrc.create("plusMinusAverage"), self.side, ("r",), "sum"
        )
        self.r_sum.attr("o3") >> self.pos_sdk.attr("r")

        # Offset Locator
        self.pos_loc = self.init_dag(mrc.Locator(), self.side, ("pos",), "loc")
        self.pos_loc_shape = self.pos_loc.get_shape()
        self.pos_loc.lhattrs("s", "v")
        self.pos_loc.set_parent(self.pos_zr)
        self.pos_loc.get_shape().attr("localScale").v = (0.1 * offset,) * 3
        self.pos_loc.attr("t") >> self.t_sum.attr("i3[0]")
        self.pos_loc.attr("r") >> self.r_sum.attr("i3[0]")
        self.last_sum = 1

        # Input Attr
        mru.add_divide_attr(self.pos_loc_shape, "input")
        self.in_nx = self.pos_loc_shape.add(ln="negX", k=True)
        self.in_px = self.pos_loc_shape.add(ln="posX", k=True)
        self.in_ny = self.pos_loc_shape.add(ln="negY", k=True)
        self.in_py = self.pos_loc_shape.add(ln="posY", k=True)
        self.in_nz = self.pos_loc_shape.add(ln="negZ", k=True)
        self.in_pz = self.pos_loc_shape.add(ln="posZ", k=True)

        # Modifiers
        def _add_modifier(driver: mrc.Attr, dir: str, color: mrc.Color):

            pn = dir[0]
            ax = dir[-1].lower()
            pnax = "{}{}".format(pn, ax)

            loc = self.init_dag(mrc.Locator(), self.side, (pnax,), "loc")
            loc.lhattrs("s", "v")
            loc.set_parent(self.pos_loc)
            loc.color = color

            locshape = loc.get_shape()
            locshape.attr("localScale").v = (offset * 0.2,) * 3

            locshape.attr("lp{}".format(ax)).v = offset * 0.1
            if pn == "n":
                locshape.attr("lp{}".format(ax)).v = -offset * 0.1

            minattr = loc.add(ln="rotMin", min=0, dv=5, k=True)
            maxattr = loc.add(ln="rotMax", min=0, dv=120, k=True)

            in_nx = locshape.add(ln="negX", k=True)
            in_px = locshape.add(ln="posX", k=True)
            in_ny = locshape.add(ln="negY", k=True)
            in_py = locshape.add(ln="posY", k=True)
            in_nz = locshape.add(ln="negZ", k=True)
            in_pz = locshape.add(ln="posZ", k=True)

            self.in_nx >> in_nx
            self.in_px >> in_px
            self.in_ny >> in_ny
            self.in_py >> in_py
            self.in_nz >> in_nz
            self.in_pz >> in_pz

            t_mult = self.init_node(
                mrc.create("multiplyDivide"), self.side, ("t", pnax), "mult"
            )
            loc.attr("t") >> t_mult.attr("i1")

            r_mult = self.init_node(
                mrc.create("multiplyDivide"), self.side, ("r", pnax), "mult"
            )
            loc.attr("r") >> r_mult.attr("i1")

            rem = self.init_node(
                mrc.create("remapValue"), self.side, (pnax,), "remap"
            )
            minattr >> rem.attr("inputMin")
            maxattr >> rem.attr("inputMax")
            rem.attr("outValue") >> t_mult.attr("i2x")
            rem.attr("outValue") >> t_mult.attr("i2y")
            rem.attr("outValue") >> t_mult.attr("i2z")
            rem.attr("outValue") >> r_mult.attr("i2x")
            rem.attr("outValue") >> r_mult.attr("i2y")
            rem.attr("outValue") >> r_mult.attr("i2z")

            t_mult.attr("o") >> self.t_sum.attr("i3[{}]".format(self.last_sum))
            r_mult.attr("o") >> self.r_sum.attr("i3[{}]".format(self.last_sum))
            self.last_sum += 1

            driver >> rem.attr("inputValue")

            return loc

        self.px_loc = _add_modifier(self.in_px, "posX", mrc.Color.red)
        self.nx_loc = _add_modifier(self.in_nx, "negX", mrc.Color.dark_red)
        self.py_loc = _add_modifier(self.in_py, "posY", mrc.Color.green)
        self.ny_loc = _add_modifier(self.in_ny, "negY", mrc.Color.dark_green)
        self.pz_loc = _add_modifier(self.in_pz, "posZ", mrc.Color.blue)
        self.nz_loc = _add_modifier(self.in_nz, "negZ", mrc.Color.dark_blue)




[docs]class VolPresRig(rig_base.BaseRig):
    def __init__(
        self,
        target: mrc.Dag,
        target_pos: mrc.Dag,
        parent: mrc.Dag,
        offset: float,
        volume_joints: Tuple[bool, bool, bool, bool, bool, bool],
        mod_name: str,
        desc: str,
        side: mrc.Side,
        joint_type: str,
    ):
        """A volume preservation rigging module.

        Args:
            target (mrc.Dag): A joint that volume preservation rig will be
                attached at and will be parented under target's parent.
            ctrl_parent (mrc.Dag):
            offset (float): Default offset value for volume joints.
            volume_joints (tuple): A tuple of 6 boolean values
                if all True values are given, the volume joints on all 6 axes
                are created.
                The order of the axes is
                (pos_x, neg_x, pos_y, neg_y, pos_z, neg_z)
            mod (str):
            desc (str):
            side (mrc.Side/None):
        """
        # Main Groups
        super(VolPresRig, self).__init__(mod_name, desc, side)

        self.meta.snap(target)
        mrc.parcons(target.get_parent(), self.meta, mo=True)
        if parent:
            self.meta.set_parent(parent)

        self.pos = self.init_dag(mrc.Null(), self.side, ("pos",), "grp")
        self.pos.snap(target)
        self.pos.set_parent(self.meta)
        mrc.pntcons(target_pos, self.pos)

        self.rot_ofst = self.init_dag(
            mrc.Null(), self.side, ("rot", "ofst"), "grp"
        )
        self.rot_ofst.snap(target)
        self.rot_ofst.set_parent(self.pos)

        self.rot = self.init_dag(mrc.Null(), self.side, ("rot",), "grp")
        self.rot.snap(target)
        self.rot.set_parent(self.rot_ofst)

        if self.side == mrc.Side.right:
            self.rot_ofst.attr("rx").v = 180

        rot_oricon = mrc.oricons(target, self.rot, mo=True)
        self.rot_blend = self.init_node(
            mrc.create("pairBlend"), self.side, ("rot",), "blend"
        )
        self.rot_blend.attr("currentDriver").v = 2
        self.rot_blend.attr("rotInterpolation").v = 1
        rot_oricon.attr("cr") >> self.rot_blend.attr("inRotate2")
        self.rot_blend.attr("weight").v = 0.5
        self.rot_blend.attr("outRotateX") >> self.rot.attr("rx")
        self.rot_blend.attr("outRotateY") >> self.rot.attr("ry")
        self.rot_blend.attr("outRotateZ") >> self.rot.attr("rz")

        self.jnt = self.init_dag(
            mru.create_joint_at(target), self.side, ("",), "jnt"
        )
        mrc.parcons(self.rot, self.jnt)
        self.jnt.set_parent(target)

        self.jnts = [self.jnt]

        # Extract rotations
        self.rot_e2q = self.init_node(
            mrc.create("eulerToQuat"), self.side, ("rot",), "e2q"
        )
        self.rot.attr("r") >> self.rot_e2q.attr("inputRotate")
        self.rot.attr("ro") >> self.rot_e2q.attr("inputRotateOrder")

        def _connect_q2e(ax: str) -> mrc.Attr:
            """Uses quatToEuler node to extract specific axis rotate value.

            Args:
                ax (str): "X", "Y" ,"Z"
            """
            q2e = self.init_node(
                mrc.create("quatToEuler"), self.side, ("r", ax.lower()), "q2e"
            )
            self.rot.attr("ro") >> q2e.attr("inputRotateOrder")
            self.rot_e2q.attr("outputQuat{}".format(ax)) >> q2e.attr(
                "inputQuat{}".format(ax)
            )
            self.rot_e2q.attr("outputQuatW") >> q2e.attr("inputQuatW")

            return q2e.attr("outputRotate{}".format(ax))

        self.rot_x = _connect_q2e("X")
        self.rot_y = _connect_q2e("Y")
        self.rot_z = _connect_q2e("Z")

        def _sep_rot_values(
            driver: mrc.Attr, ax: str
        ) -> Tuple[mrc.Attr, mrc.Attr]:
            """Uses clamp node to separate positive and negative values to
            two attributes.
            The negative value will be inverted to positive.

            Args:
                ax (str): 'X', 'Y', or 'Z'
            """
            clamp = self.init_node(
                mru.add_two_ways_clamp(driver, 0, 1000, -1000, 0),
                self.side,
                ("r{}".format(ax), "pn"),
                "clamp",
            )

            inv = self.init_node(
                mrc.create("multDoubleLinear"),
                self.side,
                ("r{}".format(ax), "n"),
                "inv",
            )
            clamp.attr("opg") >> inv.attr("i1")
            inv.attr("i2").v = -1

            return clamp.attr("opr"), inv.attr("o")

        self.out_px, self.out_nx = _sep_rot_values(self.rot_x, "x")
        self.out_py, self.out_ny = _sep_rot_values(self.rot_y, "y")
        self.out_pz, self.out_nz = _sep_rot_values(self.rot_z, "z")

        # Volume Joints
        def _create_volume_rig(ax, mult):
            pn = "p"
            if mult == -1:
                pn = "n"
            volrig = VolRig(
                mod_name="{}_{}{}".format(self.name, pn, ax),
                desc="",
                side=self.side,
                offset=offset,
                joint_type=joint_type,
            )

            # Position
            volrig.meta.snap(self.rot)
            volrig.meta.set_parent(self.rot)

            volrig.pos_loc.attr("t{}".format(ax)).v = offset * mult
            volrig.jnt.snap(volrig.pos)
            volrig.jnt.freeze()
            volrig.jnt.set_parent(self.jnt)

            mrc.parcons(volrig.pos, volrig.jnt)

            self.out_px >> volrig.in_px
            self.out_nx >> volrig.in_nx
            self.out_py >> volrig.in_py
            self.out_ny >> volrig.in_ny
            self.out_pz >> volrig.in_pz
            self.out_nz >> volrig.in_nz

            self.jnts.append(volrig.jnt)

            return volrig

        if volume_joints[0]:
            self.px_rig = _create_volume_rig("x", 1)
        if volume_joints[1]:
            self.nx_rig = _create_volume_rig("x", -1)
        if volume_joints[2]:
            self.py_rig = _create_volume_rig("y", 1)
        if volume_joints[3]:
            self.ny_rig = _create_volume_rig("y", -1)
        if volume_joints[4]:
            self.pz_rig = _create_volume_rig("z", 1)
        if volume_joints[5]:
            self.nz_rig = _create_volume_rig("z", -1)
